扩展拖拽事件支持

Example

// dom结构
// <div id="abc">
//     这里的 draggablefor 属性指向的是拖拽影响的元素，为空表示拖拽自身
//     <div draggablefor="abc"></div>
// </div>
// 可以在事件监听里取消默认行为，即 eve.preventDefault();
event.on(ele, 'dragstart', fn);
event.on(ele, 'drag', fn);
event.on(ele, 'dragend', fn);
